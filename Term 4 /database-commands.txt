برای دیدن همه دیتابیس ها
SHOW DATABASES;

برای انتخاب یک دیتابیس
USE اسم_دیتابیس;

برای دیدن تمام جدولها
SHOW TABLES;

برای دیدن جزییات جدولها
DESCRIBE اسم_جدول;

برای دیدن اطلاعات داخل جدولها
SELECT * FROM اسم_جدول;

برای ساخت دیتابیس
CREATE DATABASE اسم_دیتابیس;

برای حذف دیتابیس
DROP DATABASE اسم_دیتابیس;

 CREATE TABLE اسم_جدول (
    اسم_ستون 
);
