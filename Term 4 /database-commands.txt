برای دیدن همه دیتابیس ها
SHOW DATABASES;

برای انتخاب یک دیتابیس
USE <اسم دیتابیس>;

برای دیدن تمام جدولها
SHOW TABLES;

برای دیدن جزییات جدولها
DESCRIBE <اسم جدول>;

برای دیدن اطلاعات جدولها
SELECT * FROM <اسم جدول>;

برای ساخت دیتابیس
CREATE DATABASE <اسم دیتابیس>;

برای حذف دیتابیس
DROP DATABASE <اسم دیتابیس>;

    <اسم ستون> <>
